#### SAMPLING FUNCTIONS ####
 
def initiate_full_sampling(starting_points, make_jac, layer_num, step_size, num_steps=None, verbose=False):
    """
    inputs: starting_points - points to start walkers from
            sampling_func_down - mapping function from parameter to data space
            make_jacobian - makes jacobian to find probability of a step
            layer_num - layer to sample from (specific to neural network)
            num_steps - num of steps to attempt
            step_size - size of gaussian steps
            
    outputs: new_manifold - array of shape (num_walkers, num_steps, top_layer_size) with all steps
    
    """
    num_walkers=len(starting_points)
    if num_steps==None:
        num_steps=int(math.floor(10.**6/num_walkers)) 
        if verbose:
            print "Taking "+str(num_steps)+" steps."
    top_layer_size=starting_points.shape[1]
    manifold_new=zeros((num_walkers,num_steps,top_layer_size))
    h_pts=starting_points.copy()
    p=find_prob(make_jac, h_pts, layer_num)
    accepted=0
    acceptances=zeros(num_walkers,dtype='int')
 
    for k in range(num_steps):
        #take a step
        new_h=h_pts + make_delta(step_size, top_layer_size) 
        #accept if new probability is greater or with probability=new_p
        new_p=find_prob(make_jac, new_h, layer_num)
        qual1 = new_p>p
        qual2=(new_p/p)>[random.random() for _ in xrange(num_walkers)]
        accepted=(qual1|qual2)
        acceptances+=1*accepted
        #add the new points, or the same point again if the move was not accepted
        h_pts=accepted[:,None]*(new_h - h_pts) + h_pts
        p=accepted[:]*(new_p - p) + p
        manifold_new[:,k]=h_pts
    if verbose:       
        print str(np.sum(acceptances)*100./(num_walkers*num_steps))+"% acceptance"
    return manifold_new
 
def find_prob(make_jac, h_pts, layer_num):
    """find the probabilities of a list of points, h_pts"""
    #Calculates Jeffrey's Prior (i.e. p = product of the singular values of J)
    prob_vector=zeros(len(h_pts))
    for i , h in enumerate(h_pts):
        J=make_jac(h, layer_num)
        prob_vector[i]=prod(svdvals(J))
    return prob_vector
 
def make_delta(step_size, top_layer_size):
    """make delta_h for MC step"""
    delta_h=step_size*random.randn(top_layer_size)
    return delta_h

def find_step_size(starting_points, SdA, layer_num, num_steps, x_verbose=False, test_step_sizes=None):
    """find the optimum step size for taking a given slice"""
    if test_step_sizes==None:
        test_step_sizes=(np.arange(5)+1.)
    val=False
    while val==False:
        slopes=get_slopes(starting_points, SdA, layer_num, num_steps, test_step_sizes, x_verbose)
        if slopes.argmax()==0:
            test_step_sizes=test_step_sizes/2
        elif slopes.argmax()==4:
            test_step_sizes=test_step_sizes*2
        else:
            best_step_size=fit_slopes(test_step_sizes,slopes).item(0)
            val=True
            D=slopes.max()
    if x_verbose:
        print "The diffusion constant is: "+str(D)
    return best_step_size    
 
def get_slopes(starting_points, SdA, layer_num, num_steps, test_step_sizes, x_verbose=False):
    """find the slopes for different test step sizes"""
    slope_list=[]
    for a_step_size in test_step_sizes:
        manifold_h=initiate_full_sampling(starting_points, SdA.make_jacobian, layer_num, a_step_size, num_steps, x_verbose)
        manifold=SdA.sampling_map_down(manifold_h, layer_num)
        distance_pts=[rsqrd_vs_t(manifold,i+1) for i in xrange(len(manifold[0])-1)]
        slope=get_fit_linear(distance_pts)
        slope_list.append(slope)
    slope_list=np.asarray(slope_list)
    return slope_list
 
def rsqrd_vs_t(walks,t):
    """makes an array of r**2 distance vs time for various recorded walks"""
    totalsum=0
    for rlist in walks:
        RL=rlist[t:]-rlist[:-t]
        totalsum+=sum((RL*RL).sum(axis=1))/(len(rlist)-t)
    return totalsum/len(walks)
 
def get_fit_linear(distance_pts):
    """Find the slope of the r**2 vs t of a walk"""
    t=np.arange(len(distance_pts))
    fitfunc = lambda params, t: params[0] * t 
    errfunc = lambda p, t, distance_pts: fitfunc(p, t) - distance_pts 
 
    p1, success = leastsq(errfunc, 0.5, args = (t, distance_pts))
    return p1  
 
def fit_slopes(t, slopes):
    """fit list of slopes to a parabola to find optimum"""
    fit = scipy.polyfit(t, slopes, 2)
    return -fit[1]/(2*fit[0])
